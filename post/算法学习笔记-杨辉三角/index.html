# 生成杨辉三角
def triangle(row):
    # 使用二维数组存储内容
    trianles = []
    for j in range(row):
        triangle_inner = []
        for i in range(j):
            if i == 0 or i == (j - 1):
                triangle_inner.append(1)
            else:
                tmp = trianles[j - 1][i - 1] + trianles[j - 1][i]
                triangle_inner.append(tmp)
        trianles.append(triangle_inner)
    return trianles


sanjiao = triangle(6)

for items in sanjiao:
    for i in items:
        print(i, end='')
    print('\n')
